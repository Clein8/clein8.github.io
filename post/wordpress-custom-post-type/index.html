WordPress - Custom Post Type

WordPress 플러그인을 제작할려고 보면, Custom Post Type을 자주 접하게 된다. 가장 많이 사용하고, 그만큼 인터넷에 자료도 많고 쉽다.

  function register_item() {
    $labels = array(
      'name'               => _x( 'Items', 'textdomain' ),
    );
    $args = array(
      'labels'             => $labels,
      'public'             => true, // (bool) default: false
      'show_ui'            => true, // (bool) default: $public
      'show_in_menu'       => true, // (bool) default: $show_ui
      'query_var'          => true,
      'has_archive'        => true,
      'hierarchical'       => false, // (bool) default: false
      'supports'           => array( 'title', 'editor', 'author', 'thumbnail', 'excerpt', 'comments' ),
    );
    register_post_type( 'item', $args );
  }
  add_action( 'init', 'register_item', 0 );
