import 'package:flutter/foundation.dart' as foundation;
import 'package:flutter/cupertino.dart';
import 'package:flutter/material.dart';
import 'package:good_choice/search.dart';
import 'package:good_choice/settings/settings.dart';
import 'package:shared_preferences/shared_preferences.dart';

bool get iOS => foundation.defaultTargetPlatform == foundation.TargetPlatform.iOS;

void main() => runApp(App());

class App extends StatelessWidget {
  // This widget is the root of your application.
  @override
  Widget build(BuildContext context) {
    if (iOS) { // iOS
      return CupertinoApp(
        title: 'Good Choice',
        theme: CupertinoThemeData(
          primaryColor: CupertinoColors.destructiveRed,
          barBackgroundColor: CupertinoColors.extraLightBackgroundGray,
        ),
        home: HomePage(title: 'Cupertino App'),
      );
    } else { // Android
      return MaterialApp(
        title: 'Good Choice',
        theme: ThemeData(
          primaryColor: Colors.indigoAccent,
        ),
        home: HomePage(title: 'Meterial App'),
      );
    }
  }
}

class HomePage extends StatefulWidget {
  HomePage({Key key, this.title}) : super(key: key);

  final String title;

  @override
  _HomePageState createState() => _HomePageState();
}

class _HomePageState extends State<HomePage> {

  int _currentPage = 0;

  onTapped(int index) {
    setState(() {
      _currentPage = index;
    });
  }

  final List<Widget> tabs = [
    SearchPage(),
    SettingsPage(),
  ];

  @override
  Widget build(BuildContext context) {
    if (iOS) { // iOS
      return CupertinoTabScaffold(
        tabBar: CupertinoTabBar(items: [
          BottomNavigationBarItem(
            icon: Icon(CupertinoIcons.search), title: Text("제품 검색"),
          ),
          BottomNavigationBarItem(
            icon: Icon(CupertinoIcons.settings), title: Text("설정"),
          ),
        ]),
        tabBuilder: (context, index) {
          switch (index) {
            case 0:
              return SearchPage();
              break;

            case 1:
              return SettingsPage();
              break;

            default:
              return SearchPage();
              break;
          }
        },
      );
    } else { // Android
      return Scaffold(
        body: tabs[_currentPage],
        bottomNavigationBar: BottomNavigationBar(
          currentIndex: _currentPage,
          onTap: onTapped,
          items: [
            BottomNavigationBarItem(
              icon: Icon(Icons.search), title: Text("제품 검색"),
            ),
            BottomNavigationBarItem(
              icon: Icon(Icons.settings), title: Text("설정"),
            ),
          ],
        ),
      );
    }
  }
}
